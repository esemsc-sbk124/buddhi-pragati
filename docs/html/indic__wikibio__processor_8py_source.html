173 def _extract_names_from_infobox(self, infobox_str: str) -> List[str]:
174 """
175 Extract names from tab-separated infobox string data.
176
177 The infobox is a string in format: "name_1:value1\tname_2:value2\tother_field:value3"
178 We parse this format to extract name fields and convert to individual words.
179 """
180 names = []
181
182 if not infobox_str or not isinstance(infobox_str, str):
183 return names
184
185 try:
186 # Parse tab-separated key:value pairs
187 infobox_dict = {}
188 pairs = infobox_str.split("\t")
189
190 for pair in pairs:
191 if ":" in pair:
192 key, value = pair.split(":", 1) # Split only on first colon
193 key = key.strip()
194 value = value.strip()
195 if key and value:
196 infobox_dict[key] = value
197
198 logger.debug(f"Parsed infobox keys: {list(infobox_dict.keys())}")
199
200 # Extract sequential name fields (name_1, name_2, name_3, ...)
201 name_parts = []
202 i = 1
203 while f"name_{i}" in infobox_dict:
204 part = infobox_dict[f"name_{i}"].strip()
205 if part and part not in ["<", ">", "br"]: # Filter out HTML artifacts
206 name_parts.append(part)
207 i += 1
208
209 # If we found name parts, extract individual words
210 if name_parts:
211 for part in name_parts:
212 # Clean part of any remaining HTML-like content
213 cleaned_part = part.replace("<", "").replace(">", "").strip()
214 if self.is_single_word(cleaned_part):
215 names.append(cleaned_part)
216
217 # Also check for other common name fields
218 other_name_fields = ["full_name", "birth_name", "real_name"]
219 for field in other_name_fields:
220 if field in infobox_dict:
221 value = infobox_dict[field].strip()
222 if value and value.lower() not in ["unknown", "न/ए", "अज्ञात"]:
223 # Split multi-word names into individual words
224 words = value.split()
225 for word in words:
226 cleaned_word = (
227 word.replace("<", "").replace(">", "").strip()
228 )
229 if self.is_single_word(cleaned_word):
230 names.append(cleaned_word)
231
232 # Remove duplicates while preserving order
233 unique_names = []
234 seen = set()
235 for name in names:
236 if name not in seen and len(name) > 1: # Ensure meaningful names
237 unique_names.append(name)
238 seen.add(name)
239
240 logger.debug(f"Extracted names: {unique_names}")
241 return unique_names[:5] # Limit to 5 names per entry
242
243 except Exception as e:
244 logger.debug(f"Error parsing infobox: {e}")
245 return []

247 def _create_entry_from_name(
248 self,
249 summary: str,
250 all_names: List[str],
251 name: str,
252 source_id: str,
253 name_idx: int,
254 ) -> Optional[DatasetEntry]:
255 """Create a dataset entry by removing ALL names from summary."""
256 # Start with the original summary
257 clue = summary
258
259 # Remove ALL extracted names from the clue
260 for extracted_name in all_names:
261 clue = clue.replace(extracted_name, "").strip()
262
263 # Clean up extra whitespace
264 clue = " ".join(clue.split())
265
266 # Apply phonetic pronunciation filter
267 clue = self._remove_phonetic_pronunciations(clue)
268
269 # Check if we still have meaningful clue after removing all names
270 if len(clue) < 10: # Increased threshold since we need substantial content
271 return None
272
273 # Check clue length constraints
274 min_len = self.config.get("MIN_CLUE_LENGTH", 10)
275 max_len = self.config.get("MAX_CLUE_LENGTH", 500)
276
277 if len(clue) < min_len or len(clue) > max_len:
278 return None
279
280 # Check minimum word length for crossword suitability
281 min_word_length = self.config.get("MIN_WORD_LENGTH", 2)
282 if len(name) < min_word_length:
283 return None
284
285 # Universal sanity check - ensure answer not in clue
286 if answer_in_clue_check(clue, name):
287 logger.debug(
288 f"IndicWikiBio rejected entry: answer '{name}' found in clue '{clue[:100]}...'"
289 )
290 self.answer_in_clue_rejectionsanswer_in_clue_rejections += 1
291 return None
292
293 # Calculate quality score
294 quality_score = self.calculate_quality_score(clue, name)
295
296 # Create unique entry ID with correct format: indic_wiki_bio_{language}_{source_id}_{name_idx}
297 # We need name_idx since one source entry can generate multiple clue-answer pairs
298 entry_id = f"indic_wiki_bio_{self.language}_{source_id}_{name_idx}"
299
300 # Note: context_score will be computed later after quality filtering (deferred scoring)
301 return DatasetEntry(
302 id=entry_id,
303 clue=clue,
304 answer=name.upper(),
305 source="IndicWikiBio",
306 source_id=source_id, # Use actual source_id from dataset
307 quality_score=quality_score,
308 # context_score defaults to 0.0 and will be computed later
309 )

311 def _remove_phonetic_pronunciations(self, text: str) -> str:
312 """Remove phonetic pronunciations and transliterations from text."""
313 import re
314
315 # Patterns to remove phonetic pronunciations
316 patterns = [
317 # Latin script in parentheses: (लातिन : Franciscus [franˈtʃiskus])
318 r"\‍(लातिन\s*:\s*[^)]*\‍)",
319 r"\‍([Ll]atin\s*:\s*[^)]*\‍)",
320 # Phonetic transcriptions in brackets: [franˈtʃiskus]
321 r"\[[^\]]*ˈ[^\]]*\]", # Contains phonetic stress marks
322 r"\[[^\]]*ʃ[^\]]*\]", # Contains phonetic symbols
323 r"\[[^\]]*θ[^\]]*\]", # Contains phonetic symbols
324 r"\[[^\]]*ð[^\]]*\]", # Contains phonetic symbols
325 # Birth names in parentheses that are transliterations
326 r"\‍(\s*जन्म\s*:\s*[^)]*\‍)",
327 r"\‍([Bb]irth\s*:\s*[^)]*\‍)",
328 r"\‍([Bb]orn\s*:\s*[^)]*\‍)",
329 # Generic parenthetical content with mixed scripts
330 r"\‍([^)]*[a-zA-Z][^)]*[a-zA-Z][^)]*\‍)", # Contains multiple Latin letters
331 ]
332
333 cleaned_text = text
334 for pattern in patterns:
335 cleaned_text = re.sub(pattern, "", cleaned_text, flags=re.IGNORECASE)
336
337 # Clean up extra whitespace and punctuation
338 cleaned_text = re.sub(r"\s*;\s*", " ", cleaned_text) # Remove semicolons
339 cleaned_text = re.sub(r"\s+", " ", cleaned_text).strip()
340
341 return cleaned_text

343 def _load_batch_from_zip(
344 self, lang_code: str, batch_size: int, offset: int
345 ) -> Tuple[List[Dict], bool]:
346 """Load batch of IndicWikiBio data directly from zip files."""
347
348 # Load all data if not cached
349 if self._cached_data is None:
350 logger.info(f"Loading IndicWikiBio data for {lang_code} from zip file...")
351
352 try:
353 # Download the zip file for this language
354 zip_filename = f"data/{lang_code}_WikiBio_v1.0.zip"
355 zip_path = hf_hub_download(
356 repo_id="ai4bharat/IndicWikiBio",
357 filename=zip_filename,
358 repo_type="dataset",
359 )
360
361 # Extract and load training data
362 with zipfile.ZipFile(zip_path, "r") as zip_ref:
363 with tempfile.TemporaryDirectory() as temp_dir:
364 zip_ref.extractall(temp_dir)
365
366 # Look for training file
367 train_file = os.path.join(temp_dir, f"{lang_code}_train.jsonl")
368 if os.path.exists(train_file):
369 self._cached_data = []
370 with open(train_file, "r", encoding="utf-8") as f:
371 for line in f:
372 if line.strip():
373 entry = json.loads(line.strip())
374 self._cached_data.append(entry)
375
376 logger.info(
377 f"Loaded {len(self._cached_data)} entries from {train_file}"
378 )
379 else:
380 logger.error(f"Training file not found: {train_file}")
381 return [], False
382
383 except Exception as e:
384 logger.error(f"Failed to load data from zip for {lang_code}: {e}")
385 return [], False
386
387 # Return batch from cached data
388 if self._cached_data is None:
389 return [], False
390
391 total_entries = len(self._cached_data)
392 start_idx = offset
393 end_idx = min(offset + batch_size, total_entries)
394
395 batch = self._cached_data[start_idx:end_idx]
396 has_more = end_idx < total_entries
397
398 logger.debug(
399 f"Loaded {len(batch)} entries from zip file for {lang_code}, "
400 f"offset={offset}, has_more={has_more} (total: {total_entries})"
401 )
402
403 return batch, has_more
