17class PromptLoader:
18 """
19 Loads and manages multilingual prompt templates from JSON files.
20 """
21
22 def __init__(self):
23 self.logger = logging.getLogger("PromptLoader")
24 self.prompts_dir = Path(__file__).parent / "prompts"
25 self._prompt_cache = {}
26
27 def load_prompts(self, filename: str) -> Dict:
28 """Load prompts from a JSON file with caching."""
29 if filename in self._prompt_cache:
30 return self._prompt_cache[filename]
31
32 filepath = self.prompts_dir / filename
33 try:
34 with open(filepath, "r", encoding="utf-8") as f:
35 prompts = json.load(f)
36 self._prompt_cache[filename] = prompts
37 return prompts
38 except FileNotFoundError:
39 self.logger.error(f"Prompt file not found: {filepath}")
40 return {}
41 except json.JSONDecodeError as e:
42 self.logger.error(f"Invalid JSON in {filepath}: {e}")
43 return {}
44
45 def get_prompt(self, filename: str, key: str) -> str:
46 """Get a specific prompt (all prompts are now in English)."""
47 prompts = self.load_prompts(filename)
48 if key in prompts and "English" in prompts[key]:
49 return prompts[key]["English"]
50
51 self.logger.error(f"Prompt not found: {key} in {filename}")
52 return f"[Missing prompt: {key}]"

104 """

107 Args:

111 Returns:

127 instruction = self.loader.get_prompt(

130 prompt_parts.append(instruction)
131
132 # Add chain of thought instruction
133 if self.chain_of_thought:
134 instruction = self.loader.get_prompt(
135 "chain_of_thought.json", "chain_of_thought_instruction"
136 )
137 prompt_parts.append(instruction)
138 else:
139 instruction = self.loader.get_prompt(
140 "chain_of_thought.json", "no_chain_of_thought_instruction"
141 )
142 prompt_parts.append(instruction)
143
144 # Add main puzzle(s)
145 if isinstance(puzzle, list):
146 # Batch evaluation
147 prompt_parts.append(self._format_batch_puzzles(puzzle))
148 else:
149 # Single puzzle
150 prompt_parts.append(self._format_single_puzzle(puzzle))
151
152 # Add self-reflection instruction if configured
153 if self.self_reflection:
154 instruction = self.loader.get_prompt(
155 "reasoning_effort.json", "self_reflection_instruction"
156 )
157 prompt_parts.append(instruction)
158
159 return "\n\n".join(prompt_parts)

161 def _format_few_shot_section(self, examples: List) -> str:
162 """Format few-shot examples section."""

164 example_template = self.loader.get_prompt(
165 "few_shot_examples.json", "example_template"
166 )
167
168 examples_text = [header]
169
170 num_examples = min(len(examples), 1 if self.shot_type == "one-shot" else 3)
171
172 for i, example in enumerate(examples[:num_examples]):
173 example_text = example_template.format(
174 example_num=i + 1,
175 grid=self._format_grid_with_numbers(example),
176 clues=self._format_clues_section(example),
177 words=self._format_solution_words(example),
178 solution_grid=self._format_solution_grid(example),
179 )
180 examples_text.append(example_text)
181
182 transition = self.loader.get_prompt(
183 "few_shot_examples.json", "transition_to_puzzle"
184 )
185 examples_text.append(transition)
186
187 return "\n".join(examples_text)

233 def _format_grid_with_numbers(self, puzzle) -> str:
234 """Create a numbered grid showing where each word starts."""
235 # Create a copy of the grid to add numbers
236 display_grid = [row[:] for row in puzzle.get_grid()]
237
238 # Add number markers where words start
239 for clue in puzzle.get_clues():
240 row, col = clue.start_row, clue.start_col
241 if display_grid[row][col] == "_":
242 display_grid[row][col] = str(clue.number)
243
244 # Format for display
245 lines = []
246 for row in display_grid:
247 line = " ".join(f"{cell:>2}" for cell in row)
248 lines.append(line)
249
250 return "\n".join(lines)

252 def _format_clues_section(self, puzzle) -> str:
253 """Format the clues section with position details."""
254 instructions = []
255
256 for clue in sorted(puzzle.get_clues(), key=lambda c: (c.direction, c.number)):
257 instruction = f"""Clue number {clue.number} ({clue.direction}):
258- Clue: {clue.clue_text}
259- Length: {clue.length} letters
260- Starts at: Row {clue.start_row + 1}, Column {clue.start_col + 1}
261- Fill positions: {self._get_position_list(clue)}"""
262
263 instructions.append(instruction)
264
265 return "\n".join(instructions)

283 def _get_position_list(self, clue) -> str:
284 """Get list of grid positions for a word."""
285 positions = []
286 row, col = clue.start_row, clue.start_col
287
288 for i in range(clue.length):
289 if clue.direction == "across":
290 positions.append(f"({row + 1},{col + 1 + i})")
291 else: # down
292 positions.append(f"({row + 1 + i},{col + 1})")
293
294 return " ".join(positions)
