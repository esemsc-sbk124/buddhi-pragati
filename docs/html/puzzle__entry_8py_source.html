96 """
97 Create CrosswordPuzzleEntry from CrosswordPuzzle (evaluation format).
98
99 Args:
100 puzzle: CrosswordPuzzle instance from generator
101 context_scores: Dictionary mapping word->context_score
102 source_counts: Dictionary mapping source->count
103 generation_info: Additional generation metadata
104
105 Returns:
106 CrosswordPuzzleEntry instance
107 """
108 # Convert clues to HF format
109 clues_with_answer = {}
110 for clue in puzzle.get_clues():
111 clue_id = f"{clue.number}{clue.direction}"
112 clues_with_answer[clue_id] = {
113 "clue": clue.clue_text,
114 "answer": clue.answer,
115 "start": [clue.start_row, clue.start_col],
116 "length": clue.length,
117 "direction": clue.direction,
118 }
119
120 # Calculate mean context score
121 if context_scores:
122 mean_context = sum(context_scores.values()) / len(context_scores)
123 else:
124 mean_context = 0.0
125
126 # Calculate density from solved grid
127 solved_grid = puzzle.get_solution_grid()
128 if solved_grid:
129 filled_cells = sum(
130 1 for row in solved_grid for cell in row if cell and cell != "#"
131 )
132 total_cells = len(solved_grid) * len(solved_grid[0])
133 density = filled_cells / total_cells if total_cells > 0 else 0.0
134 else:
135 density = 0.0
136
137 # Calculate quality score (can be enhanced later)
138 quality_score = min(1.0, density * 1.2 + mean_context * 0.3)
139
140 # Get grid size - assume square grids
141 puzzle_size = puzzle.get_size()
142 if isinstance(puzzle_size, (list, tuple)):
143 grid_size = puzzle_size[0] # Use first dimension for square grids
144 else:
145 grid_size = puzzle_size
146
147 return cls(
148 id=puzzle.puzzle_id,
149 clues_with_answer=clues_with_answer,
150 empty_grid=puzzle.get_grid().tolist()
151 if hasattr(puzzle.get_grid(), "tolist")
152 else puzzle.get_grid(),
153 solved_grid=solved_grid,
154 context_score=mean_context,
155 quality_score=quality_score,
156 source_mix=source_counts,
157 grid_size=grid_size,
158 density=density,
159 word_count=len(puzzle.get_clues()),
