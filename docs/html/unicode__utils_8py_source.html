12def is_alphabetic_unicode(text: str) -> bool:
13 """
14 Check if text contains only Unicode letters (any script).
15
16 This is a Unicode-aware replacement for str.isalpha() that works
17 with all scripts including Devanagari, Bengali, Tamil, Telugu, etc.
18
19 For Indic scripts, this includes combining marks (vowel signs, etc.)
20 which are essential parts of words but have categories Mc/Mn instead of L.
21
22 Args:
23 text: Text to check
24
25 Returns:
26 True if text contains only Unicode letters and combining marks, False otherwise
27 """
28 if not text:
29 return False
30
31 # Check each character is either a Unicode letter or combining mark
32 for char in text:
33 category = unicodedata.category(char)
34 # Accept:
35 # - L* (Letter categories: Lu, Ll, Lt, Lm, Lo)
36 # - Mc (Mark, spacing combining - vowel signs in Indic scripts)
37 # - Mn (Mark, nonspacing - diacritics, anusvara, etc.)
38 if not (category.startswith("L") or category in ("Mc", "Mn")):
39 return False
40
41 return True

44def clean_unicode_text(text: Optional[str]) -> Optional[str]:
45 """
46 Clean and normalize Unicode text for crossword processing.
47
48 Args:
49 text: Input text to clean
50
51 Returns:
52 Cleaned text or None if input is invalid
53 """
54 if not text or not isinstance(text, str):
55 return None
56
57 # Normalize Unicode to NFC form (canonical composition)
58 # This ensures consistent representation of accented characters
59 cleaned = unicodedata.normalize("NFC", text.strip())
60
61 # Remove any leading/trailing whitespace
62 cleaned = cleaned.strip()
63
64 return cleaned if cleaned else None

121def is_predominantly_latin_script(text: str) -> bool:
122 """
123 Check if text is predominantly in Latin script.
124
125 This is used to filter out answers that are in Latin script
126 when we expect Indic script answers.
127
128 Args:
129 text: Text to check
130
131 Returns:
132 True if more than 50% of letter characters are Latin script
133 """

142 total_letters += 1

146 else ""

148 if char_script == "LATIN":
149 latin_letters += 1
150
151 if total_letters == 0:
152 return False

157def extract_first_sentence(text: str) -> str:
158 """
159 Extract the first sentence from text.
160
161 This function handles multiple languages and uses Unicode-aware
162 sentence boundary detection.
163
164 Args:
165 text: Input text
166
167 Returns:
168 First sentence from the text, or original text if no sentence boundary found
169 """
170 if not text:
171 return ""
172
173 text = text.strip()
174
175 # Common sentence ending punctuation across languages
176 sentence_endings = r"[।॥।।؟؍।|.!?]"
177
178 # Find first sentence ending
179 match = re.search(sentence_endings, text)
180 if match:
181 first_sentence = text[: match.end()].strip()
182 # Make sure we got a meaningful sentence (at least 10 characters)
183 if len(first_sentence) >= 10:
184 return first_sentence
185
186 # If no sentence boundary found or sentence too short, return original
187 return text
