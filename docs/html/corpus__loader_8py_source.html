78 def load_scored_corpus(
79 self, language: str, max_entries: int = None
80 ) -> List[CorpusEntry]:
81 """
82 Load corpus with cultural context scores.
83
84 Args:
85 language: Language to load (e.g., "English", "Hindi")
86 max_entries: Maximum entries to load (None for all)
87
88 Returns:
89 List of CorpusEntry with raw context scores
90 """
91 try:
92 config_name = language.lower()
93
94 logger.info(f"Loading scored corpus for {language} from {self.repo_id}")
95
96 # Load dataset with token if available
97 dataset_kwargs = {"split": "train"}
98 if self.hf_token:
99 dataset_kwargs["token"] = self.hf_token
100
101 dataset = load_dataset(self.repo_id, config_name, **dataset_kwargs)
102
103 # Process entries and create corpus with raw context scores
104 corpus_entries = []
105 raw_scores = []
106 entries_processed = 0
107
108 for entry in dataset:
109 if max_entries and entries_processed >= max_entries:
110 break
111
112 clue = entry.get("clue", "").strip()
113 answer = entry.get("answer", "").upper().strip()
114 source = entry.get("source", "unknown")
115 source_id = entry.get("source_id", "")
116 quality_score = entry.get("quality_score", 0.5)
117
118 # Basic validation
119 if not (clue and answer and len(answer) >= 2):
120 continue
121
122 # Script filtering for Indic languages - reject Latin script answers
123 if self._is_indic_language(language) and is_predominantly_latin_script(
124 answer
125 ):
126 logger.debug(
127 f"Filtered Latin script answer in {language}: {answer}"
128 )
129 continue
130
131 # Calculate or use existing context score
132 if "context_score" in entry:
133 # Use pre-computed score from dataset
134 context_score = float(entry["context_score"])
135 else:
136 context_score = 0.0
137
138 corpus_entry = CorpusEntry(
139 clue=clue,
140 answer=answer,
141 context_score=context_score,
142 source=source,
143 source_id=source_id,
144 quality_score=quality_score,
145 )
146
147 corpus_entries.append(corpus_entry)
148 raw_scores.append(context_score)
149 entries_processed += 1
150
151 if entries_processed % 100 == 0:
152 logger.debug(
153 f"Processed {entries_processed} entries for {language}"
154 )
155
156 if not corpus_entries:
157 logger.warning(f"No valid entries found for {language}")
158 return []
159
160 # Log context score distribution
161 min_score = min(raw_scores)
162 max_score = max(raw_scores)
163 mean_score = sum(raw_scores) / len(raw_scores)
164
165 logger.info(f"Context score distribution for {language}:")
166 logger.info(
167 f" Min: {min_score:.4f}, Max: {max_score:.4f}, Mean: {mean_score:.4f}"
168 )
169
170 logger.info(f"Loaded {len(corpus_entries)} scored entries for {language}")
171
172 # Cache the loaded corpus
173 self.loaded_corpus[language] = corpus_entries
174
175 return corpus_entries
176
177 except Exception as e:
178 logger.error(f"Failed to load scored corpus for {language}: {e}")
179 return []

181 def get_prioritized_corpus(
182 self, language: str, indian_threshold: float = None, max_entries: int = None
183 ) -> Tuple[List[Tuple[str, str]], Dict[str, float]]:
184 """
185 Get corpus prioritized by Indian cultural context using raw context scores.
186
187 Args:
188 language: Language to load
189 indian_threshold: Threshold for prioritizing Indian entries (applied to raw context scores)
190 max_entries: Maximum entries to load (None for all)
191
192 Returns:
193 Tuple of (prioritized_corpus, context_scores)
194 - prioritized_corpus: List of (clue, answer) pairs
195 - context_scores: Dictionary mapping answer -> context_score
196 """
197 if indian_threshold is None:
198 indian_threshold = self.config.get_float("INDIAN_CONTEXT_THRESHOLD", 0.4)
199
200 # Load scored corpus
201 scored_corpus = self.load_scored_corpus(language, max_entries)
202 if not scored_corpus:
203 logger.warning(f"No corpus loaded for {language}")
204 return [], {}
205
206 # Create return formats
207 prioritized_pairs = [(entry.clue, entry.answer) for entry in scored_corpus]
208 context_scores = {entry.answer: entry.context_score for entry in scored_corpus}
209
210 repartition = {}
211 high_context_count = 0
212
213 # Log classification statistics using raw context scores
214 for entry in scored_corpus:
215 if entry.context_score >= indian_threshold:
216 repartition[entry.source] = repartition.get(entry.source, 0) + 1
217 high_context_count += 1
218
219 # Show context score ranges for debugging
220 all_scores = [entry.context_score for entry in scored_corpus]
221
222 logger.info(f"Prioritized corpus for {language}:")
223 logger.info(f" Total entries: {len(prioritized_pairs)}")
224 logger.info(
225 f" High Indian context (>= {indian_threshold}): {high_context_count} ({high_context_count / len(scored_corpus) * 100:.1f}%)"
226 )
227 logger.info(
228 f" Context scores - Min: {min(all_scores):.4f}, Max: {max(all_scores):.4f}"
229 )
230
231 logger.info(f" Indian source repartition: {repartition}")
232 return prioritized_pairs, context_scores

256 def get_corpus_statistics(self, language: str) -> Dict[str, any]:
257 """
258 Get detailed statistics about the corpus.

264 Dictionary with comprehensive corpus statistics

276 "total_entries": len(corpus),

284 stats["context_scores"] = {
285 "min": min(context_scores),
286 "max": max(context_scores),
287 "avg": sum(context_scores) / len(context_scores),
288 "range": max(context_scores) - min(context_scores),
289 "high_context_count": sum(1 for score in context_scores if score >= 0.4),
290 "high_context_percentage": sum(
291 1 for score in context_scores if score >= 0.4
292 )
293 / len(context_scores)
294 * 100,
295 }
296
297 # Answer length distribution (for grid size planning)
298 answer_lengths = [len(entry.answer) for entry in corpus]
299 stats["answer_length"] = {
300 "min": min(answer_lengths),
301 "max": max(answer_lengths),
302 "avg": sum(answer_lengths) / len(answer_lengths),
