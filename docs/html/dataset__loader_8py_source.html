59 def load_puzzles(
60 self,
61 language: str,
62 grid_size: Optional[int] = None,
63 max_puzzles: Optional[int] = None,
64 random_sample: bool = True,
65 ) -> List[CrosswordPuzzle]:
66 """
67 Load crossword puzzles for a specific language and optional grid size.
68
69 Args:
70 language: Target language for puzzles
71 grid_size: Optional grid size filter (e.g., 10 for 10x10 grids)
72 max_puzzles: Maximum number of puzzles to load
73 random_sample: Whether to randomly sample puzzles if limiting count
74
75 Returns:
76 List of CrosswordPuzzle objects
77 """
78 self.logger.info(
79 f"Loading puzzles for language: {language}, grid_size: {grid_size}"
80 )
81
82 try:
83 # Load dataset for the language
84 cache_key = f"{language}_{grid_size or 'all'}"
85
86 if cache_key not in self._cache:
87 dataset = load_dataset(
88 self.dataset_repo,
89 name=language, # Language as config name
90 token=self.hf_token,
91 split="train", # Assuming train split contains generated puzzles
92 )
93
94 # Filter by grid size if specified
95 if grid_size is not None:
96 dataset = dataset.filter(
97 lambda example: example.get("grid_size") == grid_size
98 )
99
100 self._cache[cache_key] = dataset
101 self.logger.info(f"Loaded {len(dataset)} puzzles from HuggingFace")
102 else:
103 dataset = self._cache[cache_key]
104 self.logger.info(f"Using cached dataset with {len(dataset)} puzzles")
105
106 # Convert to list for sampling/limiting
107 puzzle_data = list(dataset)
108
109 # Apply sampling/limiting
110 if max_puzzles and len(puzzle_data) > max_puzzles:
111 if random_sample:
112 puzzle_data = random.sample(puzzle_data, max_puzzles)
113 else:
114 puzzle_data = puzzle_data[:max_puzzles]
115 self.logger.info(f"Limited to {max_puzzles} puzzles")
116
117 # Convert to CrosswordPuzzle objects
118 puzzles = []
119 for i, puzzle_dict in enumerate(puzzle_data):
120 try:
121 puzzle = self._convert_to_crossword_puzzle(puzzle_dict, i)
122 if puzzle:
123 puzzles.append(puzzle)
124 except Exception as e:
125 self.logger.warning(f"Failed to convert puzzle {i}: {e}")
126 continue
127
128 self.logger.info(f"Successfully converted {len(puzzles)} puzzles")
129 return puzzles
130
131 except Exception as e:
132 self.logger.error(f"Failed to load puzzles: {e}")
133 return []

136 def extract_number(clue):
137 """
138 Extract number from clue text.
139
140 Args:
141 clue: Clue text that may contain a number
142
143 Returns:
144 Extracted number or None if no number found
145 """
146 # Use a regex to extract the leading digits
147 match = re.match(r"(\d+)", clue)
148 if match:
149 return int(match.group(1)) # Convert the matched digits to an integer
150 return None # Return None if no number is found

152 def _convert_to_crossword_puzzle(
153 self, puzzle_dict: Dict[str, Any], index: int
154 ) -> Optional[CrosswordPuzzle]:
155 """
156 Convert HuggingFace dataset entry to CrosswordPuzzle object.
157
158 Args:
159 puzzle_dict: Dictionary from HuggingFace dataset
160 index: Index for puzzle ID generation
161
162 Returns:
163 CrosswordPuzzle object or None if conversion fails
164 """
165 try:
166 # Extract basic information using actual HF dataset field names
167 puzzle_id = puzzle_dict.get("id", f"hf_puzzle_{index}")
168 grid_size = puzzle_dict.get("grid_size", 10)
169
170 # Extract grids from HF dataset
171 empty_grid = puzzle_dict.get("empty_grid", [])
172 solved_grid = puzzle_dict.get("solved_grid", [])
173
174 # Validate grids exist
175 if not empty_grid or not solved_grid:
176 self.logger.error(f"Missing grid data for puzzle {puzzle_id}")
177 return None
178
179 # Parse clues - already structured, no JSON parsing needed
180 clues_data = puzzle_dict.get("clues", [])
181
182 # Create CrosswordClue objects and build solution_words dictionary
183 clues = []
184 solution_words = {}
185
186 for clue_dict in clues_data:
187 # Extract number from clue ID (e.g., "5across" -> 5)
188 clue_id = clue_dict.get("id", "")
189 number = self.extract_number(clue_id)
190 if number is None:
191 # Use sequential numbering if no number found in clue ID
192 number = len(clues) + 1
193
194 clue = CrosswordClue(
195 number=number,
196 direction=clue_dict["direction"],
197 clue_text=clue_dict["clue"], # HF dataset uses 'clue' field
198 answer=clue_dict["answer"],
199 start_row=clue_dict["start_row"],
200 start_col=clue_dict["start_col"],
201 length=clue_dict["length"],
202 )
203 clues.append(clue)
204
205 # Build solution_words dictionary for evaluator
206 solution_words[clue_id] = clue_dict["answer"]
207
208 # Create CrosswordPuzzle with proper grid and solution data
209 puzzle = CrosswordPuzzle(
210 puzzle_id=puzzle_id,
211 grid=empty_grid, # The puzzle to solve (empty grid)
212 solution_grid=solved_grid, # The complete solution (required by evaluator)
213 clues=clues,
214 size=(grid_size, grid_size),
215 solution_words=solution_words, # Solution dictionary (required by evaluator)
216 )
217
218 self.logger.debug(
219 f"Successfully converted puzzle {puzzle_id} with {len(clues)} clues"
220 )
221 return puzzle
222
223 except Exception as e:
224 self.logger.error(f"Error converting puzzle data: {e}")
225 return None

400 def load_puzzles_from_local(self, input_dir: str) -> List[CrosswordPuzzle]:
401 """
402 Load puzzles from local JSON files.
403
404 Args:
405 input_dir: Directory containing puzzle JSON files
406
407 Returns:
408 List of CrosswordPuzzle objects
409 """
410 input_path = Path(input_dir)
411 if not input_path.exists():
412 self.logger.error(f"Input directory does not exist: {input_dir}")
413 return []
414
415 puzzles = []
416 json_files = list(input_path.glob("*.json"))
417
418 for json_file in json_files:
419 try:
420 with open(json_file, "r", encoding="utf-8") as f:
421 puzzle_data = json.load(f)
422
423 # Convert back to CrosswordPuzzle
424 clues = []
425 for clue_data in puzzle_data["clues"]:
426 clue = CrosswordClue(
427 number=clue_data["number"],
428 direction=clue_data["direction"],
429 clue_text=clue_data["clue_text"],
430 answer=clue_data["answer"],
431 start_row=clue_data["start_row"],
432 start_col=clue_data["start_col"],
433 length=clue_data["length"],
434 )
435 clues.append(clue)
436
437 puzzle = CrosswordPuzzle(
438 puzzle_id=puzzle_data["puzzle_id"],
439 size=tuple(puzzle_data["size"]),
440 clues=clues,
441 grid=puzzle_data["grid"],
442 )
443
444 puzzles.append(puzzle)
445
446 except Exception as e:
447 self.logger.warning(f"Failed to load puzzle from {json_file}: {e}")
