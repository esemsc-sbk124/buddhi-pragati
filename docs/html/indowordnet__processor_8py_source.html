136 def load_raw_data(
137 self, batch_size: int, offset: int = 0
138 ) -> Tuple[List[Dict], bool]:
139 """Load batch of raw IndoWordNet data using memory-efficient streaming."""
140 if not self.iwn or self.total_synsets == 0:
141 logger.warning("IndoWordNet not available or no synsets initialized")
142 return [], False
143
144 # Reset iterator if we need to seek to a specific offset
145 if offset != self.current_index or self._synset_iterator is None:
146 self._reset_iterator_to_offset(offset)
147
148 # Collect batch_size entries from iterator
149 batch = []
150 entries_collected = 0
151
152 while (
153 entries_collected < batch_size and self.current_index < self.total_synsets
154 ):
155 try:
156 synset = next(self._synset_iterator)
157
158 # Get definition (gloss) for this synset
159 gloss = synset.gloss()
160 # Get list of words (lemma names) for this synset
161 lemma_names = synset.lemma_names()
162
163 batch.append(
164 {
165 "synset": synset,
166 "gloss": gloss,
167 "lemma_names": lemma_names,
168 "synset_id": str(synset.synset_id()),
169 }
170 )
171
172 entries_collected += 1
173 self.current_index += 1
174
175 except StopIteration:
176 logger.info(
177 f"Reached end of IndoWordNet synsets at position {self.current_index}"
178 )
179 break
180 except Exception as e:
181 logger.debug(
182 f"Error processing synset at position {self.current_index}: {e}"
183 )
184 self.current_index += 1
185 continue
186
187 # Determine if there's more data
188 has_more = self.current_index < self.total_synsets
189 logger.debug(
190 f"IndoWordNet batch: collected {len(batch)} entries, position {self.current_index}/{self.total_synsets}, has_more={has_more}"
191 )
192
193 return batch, has_more

262 def _process_single_item(self, item: Dict) -> List[DatasetEntry]:
263 """Process a single synset into multiple dataset entries."""
264 gloss = item.get("gloss", "").strip()
265 lemma_names = item.get("lemma_names", [])
266 synset_id = item.get("synset_id", "")
267
268 if not gloss or not lemma_names:
269 return []
270
271 # Check minimum definition length
272 min_length = self.config.get("INDOWORDNET_MIN_DEFINITION_LENGTH", 5)
273 if len(gloss) < min_length:
274 return []
275
276 # Create entries for each word (lemma) in the synset
277 entries = []
278

280 try:
281 if not word or not word.strip():

302 def _create_entry_from_word_definition(
303 self, word: str, gloss: str, synset_id: str, lemma_idx: int
304 ) -> Optional[DatasetEntry]:

306 # Clean up the gloss to make it crossword-appropriate

323 # Universal sanity check - ensure answer not in clue

338 return DatasetEntry(
339 id=entry_id,

348 def _clean_definition(self, gloss: str, word: str) -> Optional[str]:
349 """Clean definition to make it suitable as crossword clue."""
350 # Remove the word itself from definition to avoid giving away the answer
351 cleaned = gloss.replace(word, "").strip()

353 # Clean up extra whitespace
