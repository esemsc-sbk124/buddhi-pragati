70class CrosswordCandidate:
71 """Represents a candidate crossword in the population."""
72
73 def __init__(self, size: int):
74 self.size = size
75 self.grid = [["" for _ in range(size)] for _ in range(size)]
76 self.placed_words: List[WordPlacement] = []
77 self.used_clues: Set[str] = set() # Track clue texts to prevent duplicates
78 self.fitness_score: float = 0.0
79 self.density: float = 0.0
80 self.word_count: int = 0
81 self.intersection_count: int = 0
82 self.cultural_coherence: float = 0.0
83
84 def calculate_fitness(self, weights: Dict[str, float]) -> float:
85 """Calculate overall fitness score using weighted components."""
86 self.density = self._calculate_density()
87 self.word_count = len(self.placed_words)
88 self.intersection_count = self._calculate_intersections()
89 self.cultural_coherence = self._calculate_cultural_coherence()
90
91 # Normalize components
92 normalized_density = min(1.0, self.density / 0.75) # Target 75% density
93 normalized_intersections = (
94 min(1.0, self.intersection_count / (self.word_count * 0.5))
95 if self.word_count > 0
96 else 0
97 )
98 normalized_coherence = self.cultural_coherence
99
100 self.fitness_score = (
101 weights["density_weight"] * normalized_density
102 + weights["intersection_weight"] * normalized_intersections
103 + weights["cultural_coherence_weight"] * normalized_coherence
104 )
105
106 return self.fitness_score
107
108 def _calculate_density(self) -> float:
109 """Calculate grid fill density."""
110 filled_cells = sum(1 for row in self.grid for cell in row if cell != "")
111 total_cells = self.size * self.size
112 return filled_cells / total_cells if total_cells > 0 else 0.0
113
114 def _calculate_intersections(self) -> int:
115 """Calculate total number of intersections between words."""
116 intersection_count = 0
117 for i, word1 in enumerate(self.placed_words):
118 for word2 in self.placed_words[i + 1 :]:
119 if word1.intersects_at(word2):
120 intersection_count += 1
121 return intersection_count
122
123 def _calculate_cultural_coherence(self) -> float:
124 """Calculate mean cultural coherence from context scores."""
125 if not self.placed_words:
126 return 0.0
127
128 context_scores = [
129 word.context_score for word in self.placed_words if word.context_score > 0
130 ]
131 return sum(context_scores) / len(context_scores) if context_scores else 0.0

133 def clone(self) -> "CrosswordCandidate":
134 """Create a deep copy of this candidate."""
135 new_candidate = CrosswordCandidate(self.size)
136 new_candidate.grid = [row[:] for row in self.grid]
137 new_candidate.placed_words = [copy.deepcopy(word) for word in self.placed_words]
138 new_candidate.used_clues = self.used_clues.copy() # Copy the set of used clues
139 new_candidate.fitness_score = self.fitness_score
140 new_candidate.density = self.density
141 new_candidate.word_count = self.word_count
142 new_candidate.intersection_count = self.intersection_count
143 new_candidate.cultural_coherence = self.cultural_coherence
144 return new_candidate

193 def generate_puzzle_with_prioritization(

196 context_scores: Dict[str, float],

198 indian_threshold: float = None,

200 """
201 Generate crossword using memetic algorithm with cultural prioritization.
202
203 Args:
204 prioritized_corpus: Corpus sorted by Indian context (high to low)
205 context_scores: Dictionary mapping answer -> raw context_score
206 puzzle_id: Unique identifier for puzzle
207 indian_threshold: Threshold for Indian classification (for reporting)
208
209 Returns:
210 CrosswordPuzzle instance or None if generation failed
211 """
212 if indian_threshold is None:
213 indian_threshold = self.generation_config["indian_context_threshold"]
214
215 # Filter valid pairs for this grid size
216 valid_pairs = self._filter_valid_pairs(prioritized_corpus)
217 if len(valid_pairs) < self.generation_config["min_words_per_puzzle"]:
218 logger.warning(
219 f"Insufficient valid pairs ({len(valid_pairs)}) for {self.size}x{self.size} grid"
220 )
221 return None
222
223 logger.info(f"Starting memetic generation with {len(valid_pairs)} valid pairs")
224
225 # Initialize population
226 population = self._initialize_population(valid_pairs, context_scores)
227 if not population:
228 logger.error("Failed to initialize population")
229 return None
230
231 best_fitness = 0.0
232 generations_without_improvement = 0
233 max_stagnation = max(10, self.max_generations // 5)
234
235 # Evolve population
236 for generation in range(self.max_generations):
237 # Evaluate fitness for all candidates
238 for candidate in population:
239 candidate.calculate_fitness(self.fitness_weights)
240
241 # Sort by fitness (descending)
242 population.sort(key=lambda x: x.fitness_score, reverse=True)
243
244 current_best = population[0].fitness_score
245 if current_best > best_fitness:
246 best_fitness = current_best
247 generations_without_improvement = 0
248 else:
249 generations_without_improvement += 1
250
251 # Check for early termination
252 if (
253 population[0].density >= self.generation_config["target_grid_density"]
254 or generations_without_improvement >= max_stagnation
255 ):
256 break
257
258 # Log progress
259 if generation % 10 == 0:
260 logger.debug(
261 f"Generation {generation}: Best fitness={current_best:.3f}, "
262 f"Density={population[0].density:.2%}, Words={population[0].word_count}"
263 )
264
265 # Create next generation
266 new_population = []
267
268 # Keep elite (best performing candidates)
269 new_population.extend(population[: self.elite_size])
270
271 # Generate offspring through crossover and mutation
272 while len(new_population) < self.population_size:
273 if random.random() < self.crossover_rate and len(population) >= 2:
274 # Select parents using tournament selection
275 parent1 = self._tournament_selection(population, 3)
276 parent2 = self._tournament_selection(population, 3)
277
278 # Create offspring
279 offspring = self._crossover(
280 parent1, parent2, valid_pairs, context_scores
281 )
282 if offspring:
283 # Apply mutation
284 if random.random() < self.mutation_rate:
285 offspring = self._mutate(
286 offspring, valid_pairs, context_scores
287 )
288
289 # Apply local search occasionally
290 if random.random() < self.local_search_probability:
291 offspring = self._local_search(
292 offspring, valid_pairs, context_scores
293 )
294
295 new_population.append(offspring)
296 else:
297 # Generate new random candidate
298 new_candidate = self._create_random_candidate(
299 valid_pairs, context_scores
300 )
301 if new_candidate:
302 new_population.append(new_candidate)
303
304 population = new_population
305
306 # Select best candidate
307 population.sort(key=lambda x: x.fitness_score, reverse=True)
308 best_candidate = population[0]
309
310 # Check if meets minimum requirements
311 if (
312 best_candidate.density >= self.generation_config["min_acceptable_density"]
313 and best_candidate.word_count
314 >= self.generation_config["min_words_per_puzzle"]
315 ):
316 logger.info(
317 f"Generated puzzle: Density={best_candidate.density:.2%}, "
318 f"Words={best_candidate.word_count}, Fitness={best_candidate.fitness_score:.3f}"
319 )
320 else:
321 logger.warning(
322 f"Best candidate did not meet minimum requirements - returning best attempt: "
323 f"Density={best_candidate.density:.2%} (min: {self.generation_config['min_acceptable_density']:.2%}), "
324 f"Words={best_candidate.word_count} (min: {self.generation_config['min_words_per_puzzle']})"
325 )
326
327 # Always return best candidate (even if below thresholds) instead of None
328 if best_candidate.word_count < 3:
329 logger.error("Best candidate has too few words - generation failed")
330 return None
331 if best_candidate.density < 0.5:
332 logger.error("Best candidate has too low density - generation failed")
333 return None
334
335 return self._candidate_to_puzzle(best_candidate, puzzle_id)

368 """Create a random candidate using greedy word placement with cultural bias."""
369 candidate = CrosswordCandidate(self.size)
370
371 # Shuffle pairs but bias towards higher context scores (Indian cultural priority)
372 working_pairs = valid_pairs.copy()
373
374 # Use weighted random selection favoring high context scores
375 max_words = min(len(working_pairs), self.size * 2) # Reasonable limit
376 placed_count = 0
377 attempts = 0
378 max_attempts = max_words * 3
379
380 while placed_count < max_words and attempts < max_attempts and working_pairs:
381 attempts += 1
382
383 # Select word with bias towards high context scores
384 if placed_count == 0:
385 # First word: pick randomly from top context scores
386 top_candidates = working_pairs[: min(10, len(working_pairs))]

390 clue, answer = self._weighted_word_selection(
391 working_pairs, context_scores
392 )
393
394 # Try to place word
395 if placed_count == 0:
396 # Place first word in center
397 success = self._place_first_word(
398 candidate, clue, answer, context_scores.get(answer, 0.0)

400 else:
401 # Try to place intersecting word
402 success = self._place_intersecting_word(
403 candidate, clue, answer, context_scores.get(answer, 0.0)
404 )
405
406 if success:
407 placed_count += 1
408 # Remove used word
409 working_pairs = [(c, a) for c, a in working_pairs if a != answer]
410 else:
411 # Remove failed word and try next
412 working_pairs = [(c, a) for c, a in working_pairs if a != answer]

416 def _weighted_word_selection(
417 self, pairs: List[Tuple[str, str]], context_scores: Dict[str, float]
418 ) -> Tuple[str, str]:
419 """Select word with bias towards high context scores."""
420 if len(pairs) <= 5:
421 return random.choice(pairs)
422
423 # Calculate weights based on context scores
424 weights = []
425 for clue, answer in pairs:
426 context_score = context_scores.get(answer, 0.0)
427 # Exponential weighting to strongly favor high context scores
428 weight = np.exp(context_score * 3) # Scale factor 3 for strong bias
429 weights.append(weight)
430
431 # Weighted random selection
432 weights = np.array(weights)
433 probabilities = weights / weights.sum()
434
435 selected_index = np.random.choice(len(pairs), p=probabilities)
436 return pairs[selected_index]

493 new_row = placed_word.row + j

535 def _is_valid_placement(
536 self, candidate: CrosswordCandidate, placement: WordPlacement
537 ) -> bool:
538 """Check if word placement is valid (no conflicts, proper intersections, no duplicate clues)."""
539 # Check for duplicate clue text within puzzle
540 clue_normalized = placement.clue.strip().lower()
541 if clue_normalized in candidate.used_clues:
542 return False # Reject placement with duplicate clue
543
544 positions = placement.get_positions()
545
546 for i, (r, c) in enumerate(positions):
547 # Check boundaries
548 if not (0 <= r < self.size and 0 <= c < self.size):
549 return False
550
551 # Check for conflicts
552 if candidate.grid[r][c] != "" and candidate.grid[r][c] != placement.text[i]:
553 return False
554
555 # Check intersections with existing words
556 for other_word in candidate.placed_words:
557 intersections = placement.intersects_at(other_word)
558 if intersections:
559 if len(intersections) != 1: # Must have exactly one intersection
560 return False
561
562 # Verify character match at intersection
563 int_r, int_c = intersections[0]
564 self_pos = positions.index((int_r, int_c))
565 other_pos = other_word.get_positions().index((int_r, int_c))
566
567 if placement.text[self_pos] != other_word.text[other_pos]:
568 return False
569
570 return True
