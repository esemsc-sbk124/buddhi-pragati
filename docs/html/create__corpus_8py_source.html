19- Multilingual synset linking for comprehensive coverage
20- Category-organized output matching existing scorer format

51def load_indowordnet():
52 """
53 Load IndoWordNet instances for all supported languages.
54
55 Returns:
56 Dictionary mapping language names to IndoWordNet instances, or None if loading fails
57 """
58 try:
59 from pyiwn import IndoWordNet, Language
60
61 logger.info("Loading IndoWordNet for all supported languages...")
62 iwn_instances = {}
63
64 # Get all available languages
65 available_languages = [lang.value for lang in Language]
66 logger.info(f"Available languages: {available_languages}")
67
68 # Load an instance for each language
69 for lang in available_languages:
70 try:
71 logger.info(f"Loading {lang} language synsets...")
72 iwn_instances[lang] = IndoWordNet(lang=Language[lang.upper()])
73 logger.info(f"Successfully loaded {lang}")
74 except Exception as e:
75 logger.warning(f"Failed to load {lang}: {e}")
76 continue
77
78 if iwn_instances:
79 logger.info(
80 f"IndoWordNet loaded successfully for {len(iwn_instances)} languages"
81 )
82 return iwn_instances
83 else:
84 logger.error("Failed to load any IndoWordNet languages")
85 return None
86
87 except ImportError:
88 logger.error("pyiwn library not found. Install with: pip install pyiwn")
89 return None
90 except Exception as e:
91 logger.error(f"Failed to load IndoWordNet: {e}")
92 return None

154def expand_concepts_with_indowordnet(
155 seed_corpus: Dict[str, any], iwn_instances: Dict[str, any], max_depth: int = 2
156) -> Dict[str, Dict[str, Set[str]]]:
157 """
158 Expand seed concepts using IndoWordNet semantic relationships per language.
159
160 For each language-specific category in the seed corpus, uses the corresponding
161 IndoWordNet instance to find synsets and traverse hyponyms/meronyms up to max_depth
162 levels. Maintains the structured format: dict(category: dict(language: concepts)).
163
164 Args:
165 seed_corpus: Dictionary of seed concepts with structure {category: {language: [concepts]}}
166 iwn_instances: Dictionary mapping language names to IndoWordNet instances
167 max_depth: Maximum depth for semantic traversal
168
169 Returns:
170 Dictionary with structure {category: {language: expanded_concept_sets}}
171 """
172 expanded_corpus = defaultdict(lambda: defaultdict(set))
173
174 logger.info(f"Expanding corpus with max depth {max_depth}")
175 logger.info(
176 f"Using {len(iwn_instances)} IndoWordNet language instances: {list(iwn_instances.keys())}"
177 )
178
179 # Map IndoWordNet language names to our language names
180 iwn_lang_mapping = {
181 "hindi": "Hindi",
182 "bengali": "Bengali",
183 "tamil": "Tamil",
184 "telugu": "Telugu",
185 "gujarati": "Gujarati",
186 "kannada": "Kannada",
187 "malayalam": "Malayalam",
188 "marathi": "Marathi",
189 "punjabi": "Punjabi",
190 "odia": "Odia",
191 "assamese": "Assamese",
192 "sanskrit": "Sanskrit",
193 "nepali": "Nepali",
194 "urdu": "Urdu",
195 "english": "English",
196 }
197
198 for category, content in seed_corpus.items():
199 logger.info(f"Processing category: {category}")
200
201 if isinstance(content, dict):
202 # Language-specific category
203 for language, concepts in content.items():
204 if not isinstance(concepts, list):
205 continue
206
207 logger.info(
208 f"Processing {category}.{language} with {len(concepts)} seed concepts"
209 )
210 language_concepts = set()
211
212 # Find matching IndoWordNet instance for this language
213 iwn_instance = None
214 iwn_lang_name = None
215
216 # Try direct match first
217 for iwn_lang, iwn_inst in iwn_instances.items():
218 if iwn_lang.lower() == language.lower():
219 iwn_instance = iwn_inst
220 iwn_lang_name = iwn_lang
221 break
222
223 # Try mapping if no direct match
224 if iwn_instance is None:
225 for iwn_lang, mapped_lang in iwn_lang_mapping.items():
226 if (
227 mapped_lang.lower() == language.lower()
228 and iwn_lang in iwn_instances
229 ):
230 iwn_instance = iwn_instances[iwn_lang]
231 iwn_lang_name = iwn_lang
232 break
233
234 if iwn_instance is None:
235 logger.warning(
236 f"No IndoWordNet instance found for language: {language}"
237 )
238 # Just add original concepts without expansion
239 for concept in concepts:
240 language_concepts.add(concept)
241 expanded_corpus[category][language] = language_concepts
242 continue
243
244 logger.info(
245 f"Using IndoWordNet instance '{iwn_lang_name}' for language '{language}'"
246 )
247
248 # Expand concepts using the language-specific IndoWordNet instance
249 for concept in concepts:
250 # Add original concept
251 language_concepts.add(concept)
252
253 # Extract individual words from concept phrases
254 words = concept.lower().split()
255
256 for word in words:
257 # Skip common English words
258 if word in {
259 "and",
260 "or",
261 "the",
262 "a",
263 "an",
264 "is",
265 "are",
266 "with",
267 "of",
268 "in",
269 "on",
270 "at",
271 "to",
272 "for",
273 "by",
274 "from",
275 }:
276 continue
277
278 # Use only the language-specific IndoWordNet instance
279 try:
280 synsets = iwn_instance.synsets(word)
281
282 for synset in synsets[
283 :3
284 ]: # Limit to first 3 synsets per word
285 # Add synonyms from synset
286 for lemma in synset.lemma_names():
287 if (
288 lemma and len(lemma) > 2
289 ): # Valid non-trivial lemmas
290 language_concepts.add(lemma)
291
292 # Expand through hyponyms (is-a-kind-of relationships)
293 hyponyms = synset.hyponyms()
294 for hyponym in hyponyms[:5]: # Limit hyponyms
295 for lemma in hyponym.lemma_names():
296 if lemma and len(lemma) > 2:
297 language_concepts.add(lemma)
298
299 # Second level expansion
300 if max_depth >= 2:
301 second_level = hyponym.hyponyms()
302 for second_hyp in second_level[:3]:
303 for lemma in second_hyp.lemma_names():
304 if lemma and len(lemma) > 2:
305 language_concepts.add(lemma)
306
307 # Expand through meronyms (is-a-part-of relationships)
308 meronyms = synset.meronyms()
309 for meronym in meronyms[:3]: # Limit meronyms
310 for lemma in meronym.lemma_names():
311 if lemma and len(lemma) > 2:
312 language_concepts.add(lemma)
313
314 except Exception as e:
315 # Silently continue on synset errors
316 logger.debug(
317 f"Failed to process word '{word}' in {language}: {e}"
318 )
319 continue
320
321 expanded_corpus[category][language] = language_concepts
322 logger.info(
323 f"Category {category}.{language}: expanded from {len(concepts)} to {len(language_concepts)} concepts"
324 )
325 else:
326 # Global category - treat as all_languages
327 logger.info(f"Processing global category: {category}")
328 if isinstance(content, list):
329 expanded_corpus[category]["all_languages"] = set(content)
330 else:
331 logger.warning(
332 f"Unexpected content type for category {category}: {type(content)}"
333 )
334
335 return expanded_corpus

338def format_corpus_for_scorer(
339 expanded_corpus: Dict[str, Dict[str, Set[str]]],
340) -> Dict[str, Dict[str, List[str]]]:
341 """
342 Format expanded corpus to maintain structured format for indian_context_scorer.
343
344 Converts sets to lists while preserving the category -> language -> concepts structure.
345 This allows the IndianContextScorer to access language-specific keywords directly.
346
347 Args:
348 expanded_corpus: Dictionary with structure {category: {language: concept_sets}}
349
350 Returns:
351 Dictionary with structure {category: {language: concept_lists}} (scorer format)
352 """
353 formatted_corpus = {}
354
355 for category, language_dict in expanded_corpus.items():
356 formatted_corpus[category] = {}
357 for language, concepts in language_dict.items():
358 # Convert set to sorted list for deterministic output
359 formatted_corpus[category][language] = sorted(list(concepts))
360
361 return formatted_corpus
