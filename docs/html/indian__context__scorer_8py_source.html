409 def _get_reference_embeddings(self, model_name: str) -> Optional[np.ndarray]:

441 def _calculate_embedding_score(self, text: str, model_name: str) -> float:
442 """
443 Calculate semantic similarity score using embeddings.
444
445 Args:
446 text: Input text to score
447 model_name: Name of the sentence transformer model
448
449 Returns:
450 Similarity score between 0.0 and 1.0
451 """
452 model = self._load_model(model_name)
453 reference_embedding = self._get_reference_embeddings(model_name)
454
455 if model is None or reference_embedding is None:
456 logger.debug(f"Model {model_name} or reference embeddings unavailable")
457 return 0.0
458
459 try:
460 text_embedding = model.encode([text])[0]
461 # Compute cosine similarity
462 similarity = np.dot(text_embedding, reference_embedding) / (
463 np.linalg.norm(text_embedding) * np.linalg.norm(reference_embedding)
464 )
465 # Discard negative similarities (dissimilarity), keep only positive similarities
466 return max(0.0, float(similarity))
467 except Exception as e:
468 logger.debug(f"Failed to compute embedding similarity: {e}")
469 return 0.0

471 def _calculate_keyword_score(self, text: str, language: str = None) -> float:
472 """
473 Calculate language-aware keyword-based cultural context score.
474
475 Args:
476 text: Input text to analyze
477 language: Target language for language-specific keywords (optional)
478
479 Returns:
480 Keyword score between 0.0 and 1.0
481 """
482 text_lower = text.lower()
483 # Normalize unicode for better matching
484 text_normalized = unicodedata.normalize("NFKD", text_lower)
485
486 total_matches = 0
487 total_keywords = 0
488
489 # Get language-specific keywords from expanded corpus if available
490 if language and self._expanded_corpus:
491 keywords_to_check = []
492
493 # Collect language-specific keywords from all categories
494 for category, lang_dict in self._expanded_corpus.items():
495 if isinstance(lang_dict, dict):
496 # Check for exact language match first
497 if language in lang_dict:
498 keywords_to_check.extend(lang_dict[language])
499 # Check for English as fallback for better coverage
500 elif "English" in lang_dict:
501 keywords_to_check.extend(lang_dict["English"])
502 # Check for global keywords
503 elif "all_languages" in lang_dict:
504 keywords_to_check.extend(lang_dict["all_languages"])
505
506 # Process language-specific keywords
507 for keyword_phrase in keywords_to_check:
508 if isinstance(keyword_phrase, str):
509 keywords_in_phrase = keyword_phrase.lower().split()
510 total_keywords += len(keywords_in_phrase)
511
512 for keyword in keywords_in_phrase:
513 keyword_normalized = unicodedata.normalize("NFKD", keyword)
514 if keyword_normalized in text_normalized:
515 total_matches += 1
516
517 # Fallback to cultural references if no seed corpus or language-specific keywords
518 if total_keywords == 0:
519 # Check matches across cultural references (mode-based corpus)
520 for keywords in self._cultural_references.values():
521 for keyword_phrase in keywords:
522 keywords_in_phrase = keyword_phrase.lower().split()
523 total_keywords += len(keywords_in_phrase)
524
525 for keyword in keywords_in_phrase:
526 keyword_normalized = unicodedata.normalize("NFKD", keyword)
527 if keyword_normalized in text_normalized:
528 total_matches += 1
529
530 # Normalize by text length to avoid bias toward longer texts
531 word_count = len(text.split())
532 if word_count == 0 or total_keywords == 0:
533 return 0.0
534
535 # Balance between absolute matches and relative density
536 absolute_score = total_matches / total_keywords
537 density_score = total_matches / word_count
538
539 return min(1.0, (absolute_score + density_score) / 2.0)

556 def score_context(self, clue: str, answer: str, language: str) -> Dict[str, float]:
557 """
558 Calculate comprehensive Indian context score for clue-answer pair.
559
560 Args:
561 clue: Crossword clue text
562 answer: Answer word/phrase
563 language: Target language for scoring
564
565 Returns:
566 Dictionary containing individual and final scores
567 """
568 pair_text = f"{clue} {answer}"
569 strategy = self._get_scoring_strategy(language)
570
571 if strategy == "tier2":
572 # Enhanced scoring with dual models (no NER)
573 multilingual_score = self._calculate_embedding_score(
574 pair_text, self.primary_model_name
575 )
576 indic_score = self._calculate_embedding_score(
577 pair_text, self.indic_model_name
578 )
579 keyword_score = self._calculate_keyword_score(pair_text, language)
580
581 # Weighted combination for tier 2
582 final_score = (
583 self.tier2_weights["multilingual"] * multilingual_score
584 + self.tier2_weights["indic"] * indic_score
585 + self.tier2_weights["keyword"] * keyword_score
586 )
587
588 return {
589 "final_score": final_score,
590 "multilingual_score": multilingual_score,
591 "indic_score": indic_score,
592 "keyword_score": keyword_score,
593 "strategy": "tier2",
594 }
595
596 else:
597 # Standard scoring for tier 1 (no NER)
598 embedding_score = self._calculate_embedding_score(
599 pair_text, self.primary_model_name
600 )
601 keyword_score = self._calculate_keyword_score(pair_text, language)
602
603 # Weighted combination for tier 1
604 final_score = (
605 self.tier1_weights["embedding"] * embedding_score
606 + self.tier1_weights["keyword"] * keyword_score
607 )
608
609 return {
610 "final_score": final_score,
611 "embedding_score": embedding_score,
612 "keyword_score": keyword_score,
613 "strategy": "tier1",
614 }

616 def validate_cross_language_consistency(
617 self, scores_by_language: Dict[str, List[float]]
618 ) -> Dict[str, float]:
619 """
620 Validate scoring consistency across languages.
621
622 Args:
623 scores_by_language: Dictionary mapping language to list of scores
624
625 Returns:
626 Dictionary containing consistency metrics
627 """
628 if not scores_by_language:
629 return {"variance": 0.0, "std_dev": 0.0}
630
631 # Calculate mean scores per language
632 language_means = {}
633 for language, scores in scores_by_language.items():
634 if scores:
635 language_means[language] = np.mean(scores)
636
637 if not language_means:
638 return {"variance": 0.0, "std_dev": 0.0}
639
640 # Calculate variance across language means
641 overall_mean = np.mean(list(language_means.values()))
642 variance = np.var(list(language_means.values()))
643 std_dev = np.std(list(language_means.values()))
644
645 return {
646 "variance": float(variance),
647 "std_dev": float(std_dev),
648 "language_means": language_means,
649 "overall_mean": float(overall_mean),
650 }
