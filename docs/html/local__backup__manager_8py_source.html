1"""
2Local Backup Manager for Dataset Creation
3
4This module provides functionality for saving dataset entry batches locally during processing
5and loading them back for final HuggingFace upload. This enables memory-efficient processing
6while maintaining the ability to upload complete language configurations.
7
8Key Features:
9- Batch-level disk persistence during processing
10- Language-based organization of backup files
11- Automatic deduplication when loading accumulated entries
12- Cleanup functionality for post-upload maintenance
13
14Architecture:
15- Backup files stored in: configs_backup/{language}/batch_{source}_{batch_num}.json
16- JSON serialization using DatasetEntry dataclass fields
17- Answer-based deduplication across all batches for a language
18"""

57 def save_batch(
58 self, language: str, source: str, batch_num: int, entries: List[DatasetEntry]
59 ) -> str:
60 """
61 Save a batch of dataset entries to disk.
62
63 Args:
64 language: Target language (used as subdirectory)
65 source: Source name (MILU, IndicWikiBio, etc.)
66 batch_num: Batch number for this source
67 entries: List of DatasetEntry objects to save
68
69 Returns:
70 Path to the saved batch file
71 """
72 if not entries:
73 logger.debug(
74 f"No entries to save for {language} {source} batch {batch_num}"
75 )
76 return None
77
78 # Create language subdirectory
79 language_dir = self.backup_dir / language.lower()
80 language_dir.mkdir(parents=True, exist_ok=True)
81
82 # Generate batch filename
83 batch_filename = f"batch_{source}_{batch_num:03d}.json"
84 batch_file = language_dir / batch_filename
85
86 # Convert entries to JSON-serializable format
87 try:
88 entries_data = [asdict(entry) for entry in entries]
89
90 # Save to JSON file
91 with open(batch_file, "w", encoding="utf-8") as f:
92 json.dump(entries_data, f, indent=2, ensure_ascii=False)
93
94 logger.debug(f"Saved {len(entries)} entries to {batch_file}")
95 return str(batch_file)
96
97 except Exception as e:
98 logger.error(f"Failed to save batch {batch_filename} for {language}: {e}")
99 return None

101 def load_all_entries(self, language: str) -> List[DatasetEntry]:
102 """
103 Load and merge all batch files for a language with deduplication.
104
105 Args:
106 language: Target language to load entries for
107
108 Returns:
109 List of deduplicated DatasetEntry objects from all batches
110 """
111 language_dir = self.backup_dir / language.lower()
112
113 if not language_dir.exists():
114 logger.warning(f"No backup directory found for language: {language}")
115 return []
116
117 # Find all batch files
118 batch_files = list(language_dir.glob("batch_*.json"))
119 if not batch_files:
120 logger.warning(f"No batch files found for language: {language}")
121 return []
122
123 logger.info(f"Loading {len(batch_files)} batch files for {language}")
124
125 all_entries = []
126 seen_answers = set() # For deduplication
127 total_loaded = 0
128 duplicates_removed = 0
129
130 for batch_file in sorted(batch_files): # Sort for consistent ordering
131 try:
132 with open(batch_file, "r", encoding="utf-8") as f:
133 batch_data = json.load(f)
134
135 batch_entries = []
136 for entry_data in batch_data:
137 # Reconstruct DatasetEntry from dict
138 entry = DatasetEntry(**entry_data)
139
140 # Apply deduplication based on answer (case-insensitive)
141 answer_key = entry.answer.upper()
142 if answer_key not in seen_answers:
143 seen_answers.add(answer_key)
144 batch_entries.append(entry)
145 else:
146 duplicates_removed += 1
147
148 all_entries.extend(batch_entries)
149 total_loaded += len(batch_data)
150
151 logger.debug(
152 f"Loaded {len(batch_entries)}/{len(batch_data)} entries from {batch_file.name}"
153 )
154
155 except Exception as e:
156 logger.error(f"Failed to load batch file {batch_file}: {e}")
157 continue
158
159 final_count = len(all_entries)
160 logger.info(
161 f"Loaded {final_count} unique entries for {language} "
162 f"(processed {total_loaded} total, removed {duplicates_removed} duplicates)"
163 )
164
165 return all_entries

167 def cleanup_language(self, language: str) -> bool:

171 Args:
172 language: Language to clean up backup files for

174 Returns:
175 True if cleanup successful, False otherwise

178

180 logger.debug(f"No backup directory to clean for language: {language}")
181 return True
182
183 try:
184 # Count files before deletion
185 batch_files = list(language_dir.glob("batch_*.json"))
186 file_count = len(batch_files)
187
188 # Remove entire language directory
189 shutil.rmtree(language_dir, ignore_errors=True)
190
191 logger.info(
192 f"Cleaned up {file_count} backup files for language: {language}"
193 )
194 return True
195
196 except Exception as e:
197 logger.error(f"Failed to cleanup backup files for {language}: {e}")
