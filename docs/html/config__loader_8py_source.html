56 def _load_config(self):
57 """Load configuration from file."""
58 # Find project root (look for crossword_config.txt)
59 current_path = Path(__file__).parent
60 config_path = None
61
62 # Search up the directory tree
63 for _ in range(5): # Limit search depth
64 potential_path = current_path / self.config_file
65 if potential_path.exists():
66 config_path = potential_path
67 break
68 current_path = current_path.parent
69
70 if config_path is None:
71 logger.warning(f"Config file {self.config_file} not found, using defaults")
72 self._load_defaults()
73 return
74
75 logger.info(f"Loading configuration from {config_path}")
76
77 try:
78 with open(config_path, "r", encoding="utf-8") as f:
79 for line_num, line in enumerate(f, 1):
80 line = line.strip()
81
82 # Skip comments and empty lines
83 if not line or line.startswith("#"):
84 continue
85
86 # Parse key=value pairs
87 if "=" not in line:
88 logger.warning(f"Invalid config line {line_num}: {line}")
89 continue
90
91 key, value = line.split("=", 1)
92 key = key.strip()
93 value = value.strip()
94
95 # Convert value to appropriate type
96 self.config[key] = self._parse_value(value)
97
98 logger.info(f"Loaded {len(self.config)} configuration parameters")
99
100 except Exception as e:
101 logger.error(f"Failed to load config: {e}")
102 self._load_defaults()

104 def _parse_value(self, value: str) -> Union[str, int, float, bool]:
105 """Parse string value to appropriate Python type."""
106 # Handle boolean values
107 if value.lower() in ("true", "false"):
108 return value.lower() == "true"
109
110 # Handle numeric values
111 if value.replace(".", "").replace("-", "").isdigit():
112 if "." in value:
113 return float(value)
114 else:
115 return int(value)
116
117 # Return as string
118 return value
