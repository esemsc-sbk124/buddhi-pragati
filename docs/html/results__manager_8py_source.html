46 def load_experiment_results(

49 """

52 Args:

55
56 Returns:

59 pattern = f"{experiment_id}_*.json"

84 def aggregate_experiment_metrics(
85 self, experiment_results: List[Dict[str, Any]]
86 ) -> Dict[str, Any]:
87 """
88 Aggregate metrics across multiple experiment runs.
89
90 Args:
91 experiment_results: List of experiment result dictionaries
92
93 Returns:
94 Aggregated metrics with mean, std, min, max across runs
95 """
96 if not experiment_results:
97 return {}
98
99 aggregated = {
100 "experiment_count": len(experiment_results),
101 "aggregated_metrics": {},
102 "model_performance": {},
103 "statistical_summary": {},
104 }
105
106 # Extract all metric values
107 metric_values = {}
108 model_scores = {}
109
110 for result in experiment_results:
111 if "results" not in result:
112 continue
113
114 for model_name, model_result in result["results"].items():
115 if model_name not in model_scores:
116 model_scores[model_name] = {}
117
118 # Handle different result structures
119 if isinstance(model_result, dict):
120 metrics = self._extract_metrics_from_result(model_result)
121
122 for metric_name, value in metrics.items():
123 if metric_name not in metric_values:
124 metric_values[metric_name] = []
125 if metric_name not in model_scores[model_name]:
126 model_scores[model_name][metric_name] = []
127
128 metric_values[metric_name].append(value)
129 model_scores[model_name][metric_name].append(value)
130
131 # Calculate aggregated statistics
132 for metric_name, values in metric_values.items():
133 if values and all(isinstance(v, (int, float)) for v in values):
134 aggregated["aggregated_metrics"][metric_name] = {
135 "mean": np.mean(values),
136 "std": np.std(values),
137 "min": np.min(values),
138 "max": np.max(values),
139 "count": len(values),
140 }
141
142 # Calculate per-model statistics
143 for model_name, metrics in model_scores.items():
144 aggregated["model_performance"][model_name] = {}
145 for metric_name, values in metrics.items():
146 if values and all(isinstance(v, (int, float)) for v in values):
147 aggregated["model_performance"][model_name][metric_name] = {
148 "mean": np.mean(values),
149 "std": np.std(values),
150 "count": len(values),
151 }
152
153 return aggregated

155 def _extract_metrics_from_result(self, result: Dict[str, Any]) -> Dict[str, float]:
156 """Extract numeric metrics from a result dictionary."""
157 metrics = {}
158
159 # Check for summary metrics
160 if "summary_metrics" in result:
161 for key, value in result["summary_metrics"].items():
162 if isinstance(value, (int, float)):
163 metrics[key] = float(value)
164
165 # Check for direct metrics
166 if "metrics" in result:
167 for key, value in result["metrics"].items():
168 if isinstance(value, (int, float)):
169 metrics[key] = float(value)
170
171 # Handle nested structures (e.g., language family results)
172 for key, value in result.items():
173 if isinstance(value, dict) and "summary_metrics" in value:
174 for metric_key, metric_value in value["summary_metrics"].items():
175 if isinstance(metric_value, (int, float)):
176 metrics[f"{key}_{metric_key}"] = float(metric_value)
177
178 return metrics

217 def _compare_metrics_across_experiments(
218 self, all_metrics: Dict[str, Dict]
219 ) -> Dict[str, Any]:
220 """Compare the same metrics across different experiments."""
221 comparison = {}
222
223 # Find common metrics
224 all_metric_names = set()
225 for exp_metrics in all_metrics.values():
226 all_metric_names.update(exp_metrics.keys())
227
228 for metric_name in all_metric_names:
229 metric_comparison = {}
230
231 for exp_id, metrics in all_metrics.items():
232 if metric_name in metrics and "mean" in metrics[metric_name]:
233 metric_comparison[exp_id] = metrics[metric_name]["mean"]
234
235 if len(metric_comparison) > 1:
236 # Calculate relative performance
237 values = list(metric_comparison.values())
238 best_value = max(values)
239 worst_value = min(values)
240
241 comparison[metric_name] = {
242 "experiment_means": metric_comparison,
243 "best_experiment": max(
244 metric_comparison.items(), key=lambda x: x[1]
245 ),
246 "worst_experiment": min(
247 metric_comparison.items(), key=lambda x: x[1]
248 ),
249 "performance_range": best_value - worst_value,
250 "coefficient_of_variation": np.std(values) / np.mean(values)
251 if np.mean(values) > 0
252 else 0,
253 }
254
255 return comparison

388 """
389 List all available experiment types and their result counts.
390
391 Returns:
392 Dictionary mapping experiment_id to result count
393 """
394 experiments = {}
395
396 for json_file in self.results_dir.glob("*.json"):

427 def filter_results_by_priority_subset(
428 self,
429 results: Dict[str, Any],
430 priority_models: List[str] = None,
431 priority_languages: List[str] = None,
432 priority_grid_sizes: List[int] = None,
433 ) -> Dict[str, Any]:
434 """
435 Filter experiment results to only include priority subset.
436
437 Args:
438 results: Full experiment results
439 priority_models: Models to include (None = all)
440 priority_languages: Languages to include (None = all)
441 priority_grid_sizes: Grid sizes to include (None = all)
442
443 Returns:
444 Filtered results containing only priority subset
445 """
446 if "results" not in results:
447 return results
448
449 filtered_results = results.copy()
450 filtered_results["results"] = {}
451
452 # Filter by models
453 models_to_include = (
454 priority_models if priority_models else results["results"].keys()
455 )
456
457 for model_name in models_to_include:
458 if model_name not in results["results"]:
459 continue
460
461 model_data = results["results"][model_name]
462 filtered_model_data = {}
463
464 # Filter by languages
465 languages_to_include = (
466 priority_languages if priority_languages else model_data.keys()
467 )
468
469 for language in languages_to_include:
470 if language not in model_data:
471 continue
472
473 language_data = model_data[language]
474 filtered_language_data = {}
475
476 # Filter by grid sizes
477 if priority_grid_sizes:
478 grid_keys_to_include = [
479 f"grid_{size}" for size in priority_grid_sizes
480 ]
481 else:
482 grid_keys_to_include = language_data.keys()
483
484 for grid_key in grid_keys_to_include:
485 if grid_key in language_data:
486 filtered_language_data[grid_key] = language_data[grid_key]
487
488 if filtered_language_data:
489 filtered_model_data[language] = filtered_language_data
490
491 if filtered_model_data:
492 filtered_results["results"][model_name] = filtered_model_data
493
494 # Update configuration to reflect filtering
495 if "configuration" in filtered_results:
496 config = filtered_results["configuration"].copy()
497 if priority_models:
498 config["filtered_models"] = priority_models
499 if priority_languages:
500 config["filtered_languages"] = priority_languages
501 if priority_grid_sizes:
502 config["filtered_grid_sizes"] = priority_grid_sizes
503 filtered_results["configuration"] = config
504
505 return filtered_results
