118 def _compute_word_accuracy_by_direction(
119 self, parsed_words: Dict[str, str], expected_words: Dict[str, str], puzzle
120 ) -> Dict[str, float]:
121 """
122 Compute word accuracy broken down by direction (across/down).
123
124 Args:
125 parsed_words: Model's answers
126 expected_words: Ground truth answers
127 puzzle: CrosswordPuzzle for clue direction information
128
129 Returns:
130 Dict with accuracy metrics by direction
131 """
132 correct_across = 0
133 correct_down = 0
134 total_across = 0
135 total_down = 0
136 correct_total = 0
137
138 # Create direction mapping from puzzle clues
139 direction_map = {}
140 for clue in puzzle.get_clues():
141 # Format clue key consistently with parser output
142 clue_key = f"{clue.number}{clue.direction}"
143 direction_map[clue_key] = clue.direction
144
145 for clue_key, expected_answer in expected_words.items():
146 direction = direction_map.get(clue_key, "unknown")
147
148 if direction == "across":
149 total_across += 1
150 if (
151 clue_key in parsed_words
152 and parsed_words[clue_key] == expected_answer
153 ):
154 correct_across += 1
155 correct_total += 1
156 elif direction == "down":
157 total_down += 1
158 if (
159 clue_key in parsed_words
160 and parsed_words[clue_key] == expected_answer
161 ):
162 correct_down += 1
163 correct_total += 1
164 else:
165 # Handle unknown direction (shouldn't happen with proper puzzle structure)
166 self.logger.warning(f"Unknown direction for clue: {clue_key}")
167 if (
168 clue_key in parsed_words
169 and parsed_words[clue_key] == expected_answer
170 ):
171 correct_total += 1
172
173 total_words = len(expected_words)
174
175 return {
176 "word_accuracy_global": correct_total / total_words
177 if total_words > 0
178 else 0.0,
179 "word_accuracy_across": correct_across / total_across
180 if total_across > 0
181 else 0.0,
182 "word_accuracy_down": correct_down / total_down if total_down > 0 else 0.0,
183 "correct_words_across": correct_across,
184 "total_words_across": total_across,
185 "correct_words_down": correct_down,
186 "total_words_down": total_down,
187 "correct_intersections": 0, # Placeholder
188 "total_intersections": 0, # Placeholder

191 def _compute_intersection_accuracy(
192 self, parsed_grid: np.ndarray, expected_grid: np.ndarray, puzzle
193 ) -> float:
194 """
195 Compute accuracy specifically for intersection cells.
196
197 Args:
198 parsed_grid: Model's grid
199 expected_grid: Ground truth grid
200 puzzle: CrosswordPuzzle for intersection detection
201
202 Returns:
203 Intersection accuracy as float
204 """
205 intersection_positions = self._find_intersection_positions(puzzle)
206
207 if not intersection_positions:
208 self.logger.warning("No intersections found in puzzle")
209 return 0.0
210
211 correct_intersections = 0
212 for row, col in intersection_positions:
213 if (
214 row < parsed_grid.shape[0]
215 and col < parsed_grid.shape[1]
216 and parsed_grid[row, col] == expected_grid[row, col]
217 ):
218 correct_intersections += 1
219
220 return correct_intersections / len(intersection_positions)

222 def _find_intersection_positions(self, puzzle) -> Set[Tuple[int, int]]:
223 """
224 Find all intersection positions in the crossword grid.
225
226 Args:
227 puzzle: CrosswordPuzzle object
228
229 Returns:
230 Set of (row, col) tuples representing intersection positions
231 """
232 # Track all positions occupied by words
233 position_occupancy = {} # (row, col) -> list of clue IDs
234
235 for clue in puzzle.get_clues():
236 positions = []
237 for i in range(clue.length):
238 if clue.direction == "across":
239 pos = (clue.start_row, clue.start_col + i)
240 else: # down
241 pos = (clue.start_row + i, clue.start_col)
242 positions.append(pos)
243
244 # Track occupancy
245 if pos not in position_occupancy:
246 position_occupancy[pos] = []
247 position_occupancy[pos].append(f"{clue.number}{clue.direction}")
248
249 # Find intersections (positions occupied by more than one word)
250 intersections = set()
251 for pos, occupants in position_occupancy.items():
252 if len(occupants) > 1:
253 intersections.add(pos)
254
255 self.logger.debug(f"Found {len(intersections)} intersection positions")
256 return intersections
