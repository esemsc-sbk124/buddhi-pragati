417 def _calculate_summary(self, results: List[Dict]) -> Dict[str, float]:
418 """Calculate summary metrics across all results."""
419 if not results:
420 return {}
421
422 successful_results = [r for r in results if r.get("success", False)]
423
424 if not successful_results:
425 return {"success_rate": 0.0}
426
427 # Calculate averages for each metric
428 summary = {}
429
430 # Success rate
431 summary["success_rate"] = len(successful_results) / len(results)
432
433 # Average metrics across successful evaluations
434 if successful_results:
435 all_metric_names = set()
436 for result in successful_results:
437 if result.get("metrics"):
438 all_metric_names.update(result["metrics"].keys())
439
440 for metric_name in all_metric_names:
441 values = []
442 for result in successful_results:
443 if result.get("metrics") and metric_name in result["metrics"]:
444 value = result["metrics"][metric_name]
445 if isinstance(value, (int, float)):
446 values.append(value)
447
448 if values:
449 summary[f"average_{metric_name}"] = sum(values) / len(values)
450 summary[f"std_{metric_name}"] = (
451 np.std(values) if len(values) > 1 else 0.0
452 )
453
454 return summary

541 current_response = reflection_result.get("response", "")

572 def _generate_error_feedback(self, puzzle, result: Dict[str, Any]) -> List[str]:
573 """Generate error feedback for self-reflection."""
574 errors = []
575
576 # Check metrics for specific error types
577 metrics = result.get("metrics", {})
578 parsed_words = result.get("parsed_words", {})
579 expected_words = puzzle.get_solution_words()
580
581 # Word accuracy errors
582 word_accuracy = metrics.get("word_accuracy_global", 0)
583 if word_accuracy < 1.0:
584 incorrect_count = len(expected_words) - (
585 len(expected_words) * word_accuracy
586 )
587 errors.append(f"Approximately {int(incorrect_count)} words are incorrect")
588
589 # Letter accuracy errors
590 letter_accuracy = metrics.get("letter_accuracy", 0)
591 if letter_accuracy < 0.8:
592 errors.append("Many letters in the grid don't match the expected solution")
593
594 # Intersection errors
595 intersection_accuracy = metrics.get("intersection_accuracy", 0)
596 if intersection_accuracy < 0.7:
597 errors.append("Letters at word intersections don't align correctly")
598
599 # Missing or extra words
600 if len(parsed_words) != len(expected_words):
601 if len(parsed_words) < len(expected_words):
602 missing_count = len(expected_words) - len(parsed_words)
603 errors.append(
604 f"{missing_count} clue answers are missing from your response"
605 )
606 else:
607 extra_count = len(parsed_words) - len(expected_words)
608 errors.append(
609 f"You provided {extra_count} extra answers beyond the expected clues"
610 )
611
612 return errors[:3] # Limit to top 3 errors for clarity

628 def _process_and_validate_response(
629 self, response: str, puzzle_id: str
630 ) -> Optional[str]:
631 """
632 Enhanced response processing and validation for crossword evaluation.
633
634 Provides additional validation and error handling on top of backend
635 normalization to ensure robust evaluation pipeline performance.
636
637 Args:
638 response: Raw model response
639 puzzle_id: Puzzle identifier for logging
640
641 Returns:
642 Processed response string or None if processing fails
643 """
644 # Check for empty response
645 if not response or response.strip() == "":
646 self.logger.error(f"Empty response from model for puzzle {puzzle_id}")
647 return None
648
649 # Check for extremely short responses (likely incomplete)
650 if len(response.strip()) < 20:
651 self.logger.warning(
652 f"Very short response ({len(response.strip())} chars) for puzzle {puzzle_id}: {response[:50]}"
653 )
654
655 # Check for crossword-specific content
656 has_grid_marker = any(marker in response.upper() for marker in ["GRID:", "WORDS:", "ACROSS", "DOWN"])
657 has_letters = any(char.isalpha() for char in response)
658
659 if not has_grid_marker and not has_letters:
660 self.logger.warning(
661 f"Response lacks crossword structure markers for puzzle {puzzle_id}"
662 )
663
664 # Log response characteristics for debugging
665 self.logger.debug(
666 f"Response processing for {puzzle_id}: "
667 f"length={len(response)}, has_grid_marker={has_grid_marker}, "
668 f"has_letters={has_letters}"
669 )
670
671 # Return processed response (normalized by backend)
672 return response.strip()
