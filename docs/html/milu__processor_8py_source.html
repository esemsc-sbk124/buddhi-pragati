146 def load_raw_data(
147 self, batch_size: int, offset: int = 0

149 """Load batch of raw MILU data using streaming to avoid loading entire dataset."""
150 if not hasattr(self, "_stream_initialized"):
151 logger.info(f"Initializing MILU streaming dataset for {self.language}...")
152 try:

154 self.dataset = load_dataset(
155 "ai4bharat/MILU",
156 name=self.language,
157 split="test",
158 streaming=True, # This is the key change - streaming mode
159 )
160 self._stream_iterator = iter(self.dataset)
161 self._current_position = 0
162 self._stream_initialized = True
163 logger.info(f"Initialized MILU streaming dataset for {self.language}")
164 except Exception as e:
165 logger.error(
166 f"Failed to load MILU streaming dataset for {self.language}: {e}"
167 )
168 return [], False
169
170 # Skip to desired offset if needed
171 while self._current_position < offset:
172 try:
173 next(self._stream_iterator)
174 self._current_position += 1
175 except StopIteration:
176 logger.info(
177 f"Reached end of MILU dataset at position {self._current_position}"
178 )
179 return [], False
180
181 # Collect batch_size entries
182 batch = []
183 entries_collected = 0
184
185 while entries_collected < batch_size:
186 try:
187 item = next(self._stream_iterator)
188 batch.append(dict(item))
189 entries_collected += 1
190 self._current_position += 1
191 except StopIteration:
192 logger.info(
193 f"Reached end of MILU dataset at position {self._current_position}"
194 )
195 break
196
197 # Return batch and whether more data is available
198 # More intelligent has_more detection to prevent early termination
199 has_more = entries_collected >= min(
200 batch_size * 0.5, 50
201 ) # At least 50% of requested or 50 entries minimum
202 logger.debug(f"MILU batch: collected {len(batch)} entries, has_more={has_more}")
203
204 return batch, has_more

244 def _process_single_item(self, item: Dict) -> DatasetEntry:
245 """Process a single MILU item into a dataset entry."""
246 question = item.get("question", "").strip()
247 if not question:
248 return None
249
250 # Check if this is a contextual question that needs options
251 if self.config.get("MILU_FILTER_CONTEXTUAL_QUESTIONS", True):
252 if self._is_contextual_question(question):
253 return None
254
255 # Get the correct answer (target is like 'option1', 'option2', etc.)
256 target = item.get("target", "")
257 if not target or not isinstance(target, str) or not target.startswith("option"):
258 return None
259
260 # Extract answer directly using the target field name
261 answer = item.get(target, "").strip()
262 if not answer:
263 return None
264
265 # For Indic languages, filter out entries where options/answer are in Latin script
266 if self.language != "English":
267 # Check if answer is predominantly Latin script (should be filtered out for Indic languages)
268 if is_predominantly_latin_script(answer):
269 return None
270
271 # Also check all options to see if any are Latin script
272 for i in range(1, 5): # MILU has up to 4 options
273 option_key = f"option{i}"
274 if option_key in item:
275 option_value = item.get(option_key, "").strip()
276 if option_value and is_predominantly_latin_script(option_value):
277 return None
278
279 # Check if answer is single word
280 if not self.is_single_word(answer):
281 return None
282
283 # Clean up the question to make it more crossword-like
284 cleaned_question = self._clean_question(question)
285 if not cleaned_question:
286 return None
287
288 # Universal sanity check - ensure answer not in clue
289 if answer_in_clue_check(cleaned_question, answer):
290 logger.debug(
291 f"MILU rejected entry: answer '{answer}' found in clue '{cleaned_question[:100]}...'"
292 )
293 self.answer_in_clue_rejectionsanswer_in_clue_rejections += 1
294 return None
295
296 # Check minimum word length for crossword suitability
297 min_word_length = self.config.get("MIN_WORD_LENGTH", 2)
298 if len(answer) < min_word_length:
299 return None
300
301 # Create dataset entry with new ID format: milu_{language}_{batch_number}_{count_in_batch}
302 entry_id = f"milu_{self.language}_{self.current_batch_number}_{self.valid_count_in_batch}"
303
304 quality_score = self.calculate_quality_score(cleaned_question, answer)
305
306 # Note: context_score will be computed later after quality filtering (deferred scoring)
307 return DatasetEntry(
308 id=entry_id,
309 clue=cleaned_question,
310 answer=answer.upper(),
311 source="MILU",
312 source_id="N/A", # As requested, MILU entries have no source ID
313 quality_score=quality_score,
314 # context_score defaults to 0.0 and will be computed later
315 )

317 def _is_contextual_question(self, question: str) -> bool:
318 """Check if question requires contextual knowledge or multiple choice options."""
319 for pattern in self.contextual_patterns:
320 if pattern.search(question):
321 return True
322 return False

324 def _clean_question(self, question: str) -> str:
325 """Clean question to make it more suitable as a crossword clue."""
326 # Remove MCQ-specific phrases
327 cleaned = question
328
329 # Remove common MCQ introductions
330 mcq_phrases = [
331 r"^\s*(?:Q\.?\s*\d+\.?\s*)?", # Question numbers
332 r"\s*\‍([A-D]\‍)\s*$", # Option markers at end
333 r"\s*[A-D][\.\‍)]\s*$", # Option markers at end
334 ]
335
336 for phrase in mcq_phrases:
337 cleaned = re.sub(phrase, "", cleaned, flags=re.IGNORECASE)
338
339 # Clean up whitespace
340 cleaned = " ".join(cleaned.split())
341
342 # Check length constraints
343 min_len = self.config.get("MIN_CLUE_LENGTH", 10)
344 max_len = self.config.get("MAX_CLUE_LENGTH", 500)
345
346 if len(cleaned) < min_len or len(cleaned) > max_len:
347 return None
348
349 return cleaned
